Open Dialog! Close Dialog!

Advanced Example

document.getElementById('ny-btn').onclick = function() {
    setDialog("open", {
        title: "The Dialog Box Title",
        width: 400,
        height: 200,
        content: "Hello!",
        buttons: {
            "Delete": function() {
                setDialog("open", {
                    title: "Confirmation",
                    content: "Are you sure?",
                    overlay: true,
                    buttons: {
                        "Yes": function() {
                            setDialog("close");
                        },
                        "No": function() {
                            alert("Canceled!");
                            setDialog("close");
                        }
                    }
                });
            }
        }
    });
};
